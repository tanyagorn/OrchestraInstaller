Interface Message

 

package emds.epi.decl.server.message;

 
public interface Message {
     
  /**
     * Returns the unique identifier of this message.
     */ 
  String getMessageID();
 
  /**
     * Returns an map of the message properties
     */ 
  MessageProperties getProperties() throws EpiBaseException;
     
  /**
     * Computes the "MD5" fingerprint of the message
     * @throws EpiBaseException
     */ 
  String getFingerPrint() throws EpiBaseException;
 
  /**
     * Computes the required hash fingerprint of the message
     * @throws EpiBaseException
     */ 
  String getFingerPrint(String algorithm, boolean asHex) throws EpiBaseException;
}

 

Interface MessageProperties

 
package emds.epi.decl.server.message;

 

public interface MessageProperties {

 
  /**

   * Returns a map - view on the given message properties. The returned
     * map is defined as unmodifiable. The returned map is not changeable.
     */
  Map<String,String> getUnmodifieableMap();
 
  /**
     * Append the given key value pairs in the given map to this message properties
     */
  void append( Map<String,String> mapToAppend ) throws EpiBaseException;
 
  /**
     * Store/modify a new entry in the MessageProperties
     * @throws EpiBaseException
     */
  void put( String key, String value ) throws EpiBaseException;
 
  /**
     * Checks whether the message properties contain an entry named with the given key
     */
  boolean containsKey( String key );
 
  /**
     * Return the value of a given message property
     */
  String get( String key );
 
  /**
     * Remove a named property from the message properties
     */
  void remove( String key ) throws EpiBaseException;
 
}

 
Interface MessageList

 
package emds.epi.decl.server.message;
 
public interface MessageList extends Serializable, QualityOfServiceAdapter<MessageList> {
 

  /**
   * Returns the unique qualified name of the message list
   */
  QName getMsgListIdentifier();
 

  /**
   * Returns true if the given message list is an anonymous message list
   */

   boolean isAnonymous();
 

  /**
   * Returns the number of messages stored in this message list
   */
   int size() throws EpiBaseException;
 

  /**
   * Adds the given message to the message list.
   */
   void add( Message message ) throws EpiBaseException;  
 

  /**
   * Adds the given message to the message list. Additional information
   * for the message list entry is provided by the given properties map
   */
   void add( Message message, Map<String,String> props) throws EpiBaseException;
 

  /**
   * Adds additional information for the message list entry at the given position.
   *
   * @param index position of the message within the list
   * @param key property name
   * @param value property value
   */
   void addProperty( int index, String key, String value ) throws EpiBaseException;
 

  /**
   * Returns a message from the message list at the given position
   */
  Message get(int index) throws EpiBaseException;
 

  /**
   * Returns the list properties of the message at the given position
   */
  Map<String,String> getProperties (int index) throws EpiBaseException;  
 

  /**
   * Returns true if the given message list is empty otherwise false
   */
   boolean isEmpty () throws EpiBaseException;

 

  /**
   * returns the index of the message or -1
   */
   int indexOf( Message message ) throws EpiBaseException;
 

  /**
   * returns the index of the first entry having the given property or -1
   */
   int indexOf( String propName, String value ) throws EpiBaseException;
 

  /*
   * removes the message at the given position from the list.
   */
   void deleteEntry( int index ) throws EpiBaseException;
     

  /**
   * search all entries matching the given search expression.
   */
  List<MessageListEntry> searchAll( SearchPredicate searchExpression ) throws EpiBaseException;

 

 /**

   * Returns an iterator for the given message list

   */

   OrcIterator<MessageListEntry> getEntries() throws EpiBaseException;

 

  /**

   * Delete all entries from the message list

   */        

   void deleteAll( List<MessageListEntry> entries ) throws EpiBaseException;

 

  /**

   * Add all of the entries to the end of the list

   */

   void addAll( List<MessageListEntry> entries )  throws EpiBaseException;
}

 

Interface MessageListEntry

 
package emds.epi.decl.server.message;
 
public interface MessageListEntry {

 

  /**

   * Returns the properties of the this message list entry

   */

   Map<String,String> getProperties() throws EpiBaseException;

 

  /**

   * Returns the message of this message list entry

   */

   Message getMessage() throws EpiBaseException;

 

  /**

   * Deletes this entry from the message list.

   */

   void delete() throws EpiBaseException;

 

  /**

   * Returns the identifier of the message from this entry or null if the message could not be retrieved.

   */

   String getMessageIdentifier();

}
