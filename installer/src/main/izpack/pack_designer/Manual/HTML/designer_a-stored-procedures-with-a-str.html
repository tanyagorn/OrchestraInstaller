import java.sql.Connection;
import java.sql.CallableStatement;
import java.sql.ResultSet;
import java.sql.Types;
 
/********************************************************************************
* This sample explains the invocation of a stored function that returns  
* a value of type varchar
*  
*   Step 1) Obtain a database connection from orchestra
*   Step 2) Prepare the invocation of the stored procedure  
*           (Note the syntax { ? = .... } ) This indicates that parameter  
*           1 is of type out
*   Step 3) A return value is prepared as a out-Parameter at position 1
*   Step 4) Bind the parameters to the stored procedure
*   Step 5) Execute the stored procedure
*   Step 6) Get the result from the stored procedure
*   Step 7) Close the statement
*  
* Note: A commit is done automatically
*
*******************************************************************************/
 
/*
* Step 1) Obtain a database connection from orchestra
*/
 
Connection     connection =
soffico.customerapi.landscape.DatabaseAdapter.getConnection(
this.getScenarioIdentifier(), "env_testdatabase" );
 
/*
* Step 2) Prepare the invocation of the stored procedure.
* Note the syntax { ? = .... } )  
* This indicates that parameter 1 is an output parameter that receives the result
*/
 
CallableStatement   statement  = connection.prepareCall( "{ ? = call simpleConcat(?,?,?)}" );
 
try {
 
  /*
   * Step 3) A return value is prepared as a out-Parameter at position 1
   */
  statement.registerOutParameter(1, Types.VARCHAR);
 
/*
  * Step 4) Bind the parameters to the stored procedure
  */
   
 statement.setString( 2, ident );
 statement.setString( 3, col1 );
 statement.setString( 4, col2 );
 
/*
  * Step 5) Execute the stored procedure
  */
   
  statement.execute();
 
/*
  * Step 6) Get the result from the stored procedure
  */
 
 result = statement.getString( 1 );
}
finally {
 
/*
  * Step 7) Close the statement
  */
 
if( statement != null ) {
   statement.close();
 }
}
